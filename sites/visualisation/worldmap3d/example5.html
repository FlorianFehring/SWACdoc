Example 5: Useing worldmap3d and geolocation components together

VIEW-Requestor 

document.addEventListener('swac_components_complete', function () {
  window.swac.reactions.addReaction(function(requestors){
    let worldmap3dComp = requestors['worldmap3d'].swac_comp;
    requestors['worldmap3d_geolocate'].swac_comp.options.onLocateFunctions[0] = worldmap3dComp.onUserLocation.bind(worldmap3dComp);
  },'worldmap3d','worldmap3d_geolocate');
});
